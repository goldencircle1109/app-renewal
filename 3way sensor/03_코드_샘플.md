### 1-3. 활동 전환 감지 등록

```kotlin
// ActivityRecognitionHelper.kt

import com.google.android.gms.location.*

class ActivityRecognitionHelper(private val context: Context) {

    private val client = ActivityRecognition.getClient(context)

    // 감지할 활동 전환 목록 설정
    fun startTracking() {
        val transitions = listOf(
            // 걷기 시작/종료
            ActivityTransition.Builder()
                .setActivityType(DetectedActivity.WALKING)
                .setActivityTransition(ActivityTransition.ACTIVITY_TRANSITION_ENTER)
                .build(),
            ActivityTransition.Builder()
                .setActivityType(DetectedActivity.WALKING)
                .setActivityTransition(ActivityTransition.ACTIVITY_TRANSITION_EXIT)
                .build(),
            // 뛰기 시작/종료
            ActivityTransition.Builder()
                .setActivityType(DetectedActivity.RUNNING)
                .setActivityTransition(ActivityTransition.ACTIVITY_TRANSITION_ENTER)
                .build(),
            ActivityTransition.Builder()
                .setActivityType(DetectedActivity.RUNNING)
                .setActivityTransition(ActivityTransition.ACTIVITY_TRANSITION_EXIT)
                .build(),
            // 자전거 시작/종료
            ActivityTransition.Builder()
                .setActivityType(DetectedActivity.ON_BICYCLE)
                .setActivityTransition(ActivityTransition.ACTIVITY_TRANSITION_ENTER)
                .build(),
            ActivityTransition.Builder()
                .setActivityType(DetectedActivity.ON_BICYCLE)
                .setActivityTransition(ActivityTransition.ACTIVITY_TRANSITION_EXIT)
                .build()
        )

        val request = ActivityTransitionRequest(transitions)

        // PendingIntent로 백그라운드에서도 수신 가능
        val intent = Intent(context, ActivityTransitionReceiver::class.java)
        val pendingIntent = PendingIntent.getBroadcast(
            context, 0, intent,
            PendingIntent.FLAG_UPDATE_CURRENT or PendingIntent.FLAG_MUTABLE
        )

        client.requestActivityTransitionUpdates(request, pendingIntent)
            .addOnSuccessListener {
                Log.d("Activity", "활동 인식 등록 성공")
            }
            .addOnFailureListener { e ->
                Log.e("Activity", "활동 인식 등록 실패: ${e.message}")
            }
    }

    fun stopTracking() {
        val intent = Intent(context, ActivityTransitionReceiver::class.java)
        val pendingIntent = PendingIntent.getBroadcast(
            context, 0, intent,
            PendingIntent.FLAG_UPDATE_CURRENT or PendingIntent.FLAG_MUTABLE
        )
        client.removeActivityTransitionUpdates(pendingIntent)
    }
}
```

```kotlin
// ActivityTransitionReceiver.kt

class ActivityTransitionReceiver : BroadcastReceiver() {

    override fun onReceive(context: Context, intent: Intent) {
        if (ActivityTransitionResult.hasResult(intent)) {
            val result = ActivityTransitionResult.extractResult(intent) ?: return

            for (event in result.transitionEvents) {
                val activityName = getActivityName(event.activityType)
                val transitionType = getTransitionName(event.transitionType)

                Log.d("Activity", "활동 변화: $activityName → $transitionType")

                // 활동 시작 시 기록 시작
                if (event.transitionType == ActivityTransition.ACTIVITY_TRANSITION_ENTER) {
                    startActivityRecord(context, event.activityType)
                }
                // 활동 종료 시 기록 저장
                else if (event.transitionType == ActivityTransition.ACTIVITY_TRANSITION_EXIT) {
                    endActivityRecord(context, event.activityType)
                }
            }
        }
    }

    private fun getActivityName(type: Int): String {
        return when (type) {
            DetectedActivity.WALKING -> "걷기"
            DetectedActivity.RUNNING -> "뛰기"
            DetectedActivity.ON_BICYCLE -> "자전거"
            DetectedActivity.STILL -> "정지"
            else -> "알 수 없음"
        }
    }

    private fun getTransitionName(type: Int): String {
        return when (type) {
            ActivityTransition.ACTIVITY_TRANSITION_ENTER -> "시작"
```

### 2-2. 활동 인식 매니저

```swift
// ActivityRecognitionManager.swift

import CoreMotion
import Foundation

class ActivityRecognitionManager {

    private let motionActivityManager = CMMotionActivityManager()
    private let pedometer = CMPedometer()

    // 현재 활동 상태
    var currentActivity: ActivityType = .unknown
    var onActivityChanged: ((ActivityType) -> Void)?

    enum ActivityType: String {
        case walking = "WALKING"
        case running = "RUNNING"
        case cycling = "CYCLING"
        case stationary = "STATIONARY"
        case unknown = "UNKNOWN"
    }

    // 활동 인식 가능 여부 확인
    func isAvailable() -> Bool {
        return CMMotionActivityManager.isActivityAvailable()
    }

    // 실시간 활동 모니터링 시작
    func startTracking() {
        guard isAvailable() else {
            print("이 기기는 활동 인식을 지원하지 않습니다")
            return
        }

        motionActivityManager.startActivityUpdates(to: .main) { [weak self] activity in
            guard let activity = activity else { return }

            let newActivity = self?.classifyActivity(activity) ?? .unknown

            // 활동이 변경되었을 때만 처리
            if newActivity != self?.currentActivity {
                let previous = self?.currentActivity ?? .unknown
                self?.currentActivity = newActivity
                self?.onActivityChanged?(newActivity)

                print("활동 변경: \(previous.rawValue) → \(newActivity.rawValue)")
                print("  신뢰도: \(self?.confidenceString(activity.confidence) ?? "?")")
            }
        }

        print("활동 인식 시작됨")
    }

    // 활동 모니터링 중지
    func stopTracking() {
        motionActivityManager.stopActivityUpdates()
        print("활동 인식 중지됨")
    }

    // CMMotionActivity → 우리 활동 유형으로 변환
    private func classifyActivity(_ activity: CMMotionActivity) -> ActivityType {
        // 신뢰도가 low이면 무시
        if activity.confidence == .low {
            return currentActivity // 기존 활동 유지
        }

        // 우선순위: cycling > running > walking > stationary
        if activity.cycling {
            return .cycling
        } else if activity.running {
            return .running
        } else if activity.walking {
            return .walking
        } else if activity.stationary {
            return .stationary
        }

        return .unknown
    }

    // 신뢰도를 문자열로 변환
    private func confidenceString(_ confidence: CMMotionActivityConfidence) -> String {
        switch confidence {
        case .low: return "낮음"
        case .medium: return "보통"
        case .high: return "높음"
        @unknown default: return "?"
        }
    }

    // 과거 활동 기록 조회 (최대 7일)
    func queryHistory(from: Date, to: Date, completion: @escaping ([CMMotionActivity]) -> Void) {
        motionActivityManager.queryActivityStarting(
            from: from,
            to: to,
            to: .main
        ) { activities, error in
            if let error = error {
                print("히스토리 조회 실패: \(error.localizedDescription)")
                completion([])
                return
            }
            completion(activities ?? [])
        }
    }

    // 걸음수 모니터링 시작
    func startPedometerUpdates(completion: @escaping (Int, Double) -> Void) {
        guard CMPedometer.isStepCountingAvailable() else { return }

        pedometer.startUpdates(from: Date()) { data, error in
            guard let data = data, error == nil else { return }

            let steps = data.numberOfSteps.intValue
            let distance = data.distance?.doubleValue ?? 0

            completion(steps, distance)
        }
    }
}
```

### 2-3. 사용 예시

```swift
// ViewController.swift

class ActivityViewController: UIViewController {

    let activityManager = ActivityRecognitionManager()

    override func viewDidLoad() {
        super.viewDidLoad()

        // 활동 변경 콜백 설정
        activityManager.onActivityChanged = { [weak self] activity in
            DispatchQueue.main.async {
                self?.updateUI(activity: activity)
            }
        }

        // 활동 인식 시작
        activityManager.startTracking()

        // 걸음수 모니터링 시작
        activityManager.startPedometerUpdates { steps, distance in
            print("걸음수: \(steps), 거리: \(distance)m")
        }
    }

    private func updateUI(activity: ActivityRecognitionManager.ActivityType) {
        switch activity {
        case .walking:
            activityLabel.text = "걷는 중"
            activityIcon.image = UIImage(systemName: "figure.walk")
        case .running:
            activityLabel.text = "뛰는 중"
            activityIcon.image = UIImage(systemName: "figure.run")
        case .cycling:
            activityLabel.text = "자전거 타는 중"
            activityIcon.image = UIImage(systemName: "bicycle")
        case .stationary:
            activityLabel.text = "정지"
            activityIcon.image = UIImage(systemName: "pause.circle")
        case .unknown:
            activityLabel.text = "감지 중..."
            activityIcon.image = UIImage(systemName: "questionmark.circle")
        }
    }

    deinit {
        activityManager.stopTracking()
    }
}
```

```javascript
// ActivityTracker.js

import React, { useState, useEffect } from 'react';
import { View, Text, StyleSheet } from 'react-native';
import ActivityRecognition from 'react-native-activity-recognition';

const ACTIVITY_LABELS = {
  WALKING: '걷기',
  RUNNING: '뛰기',
  ON_BICYCLE: '자전거',
  STILL: '정지',
  UNKNOWN: '감지 중...',
};

const ACTIVITY_ICONS = {
  WALKING: '🚶',
  RUNNING: '🏃',
  ON_BICYCLE: '🚴',
  STILL: '⏸️',
  UNKNOWN: '❓',
};

export default function ActivityTracker() {
  const [currentActivity, setCurrentActivity] = useState('UNKNOWN');
  const [confidence, setConfidence] = useState(0);

  useEffect(() => {
    // 활동 인식 구독 시작
    const subscription = ActivityRecognition.subscribe((detectedActivities) => {
      // 신뢰도가 가장 높은 활동 선택
      const topActivity = detectedActivities.sorted[0];

      if (topActivity && topActivity.confidence >= 75) {
        setCurrentActivity(topActivity.type);
        setConfidence(topActivity.confidence);
      }
    });

    // 1초 간격으로 감지 (Android에서 사용)
    ActivityRecognition.start(1000);

    // 컴포넌트 언마운트 시 정리
    return () => {
      subscription.remove();
      ActivityRecognition.stop();
    };
  }, []);

  return (
    <View style={styles.container}>
      <Text style={styles.icon}>
        {ACTIVITY_ICONS[currentActivity] || '❓'}
      </Text>
      <Text style={styles.activity}>
        {ACTIVITY_LABELS[currentActivity] || '감지 중...'}
      </Text>
      <Text style={styles.confidence}>
        신뢰도: {confidence}%
      </Text>
    </View>
  );
}

const styles = StyleSheet.create({
  container: {
    alignItems: 'center',
    justifyContent: 'center',
    padding: 20,
  },
  icon: {
    fontSize: 64,
  },
  activity: {
    fontSize: 24,
    fontWeight: 'bold',
    marginTop: 10,
  },
  confidence: {
    fontSize: 16,
    color: '#666',
    marginTop: 5,
  },
});
```

```dart
// activity_tracker.dart

import 'package:flutter/material.dart';
import 'package:flutter_activity_recognition/flutter_activity_recognition.dart';
import 'dart:async';

class ActivityTracker extends StatefulWidget {
  @override
  _ActivityTrackerState createState() => _ActivityTrackerState();
}

class _ActivityTrackerState extends State<ActivityTracker> {
  final _activityRecognition = FlutterActivityRecognition.instance;
  StreamSubscription<Activity>? _subscription;

  String _currentActivity = '감지 중...';
  IconData _currentIcon = Icons.help_outline;

  @override
  void initState() {
    super.initState();
    _startTracking();
  }

  Future<void> _startTracking() async {
    // 권한 확인
    PermissionRequestResult result =
        await _activityRecognition.checkPermission();

    if (result == PermissionRequestResult.PERMANENTLY_DENIED) {
      print('활동 인식 권한이 영구 거부되었습니다');
      return;
    }

    if (result != PermissionRequestResult.GRANTED) {
      result = await _activityRecognition.requestPermission();
      if (result != PermissionRequestResult.GRANTED) {
        print('활동 인식 권한이 거부되었습니다');
        return;
      }
    }

    // 활동 스트림 구독
    _subscription = _activityRecognition.activityStream.listen(
      (Activity activity) {
        setState(() {
          switch (activity.type) {
            case ActivityType.WALKING:
              _currentActivity = '걷기';
              _currentIcon = Icons.directions_walk;
              break;
            case ActivityType.RUNNING:
              _currentActivity = '뛰기';
              _currentIcon = Icons.directions_run;
              break;
            case ActivityType.ON_BICYCLE:
              _currentActivity = '자전거';
              _currentIcon = Icons.directions_bike;
              break;
            case ActivityType.STILL:
              _currentActivity = '정지';
              _currentIcon = Icons.pause_circle;
              break;
            default:
              _currentActivity = '감지 중...';
              _currentIcon = Icons.help_outline;
          }
        });
        print('활동 감지: ${activity.type}, 신뢰도: ${activity.confidence}');
      },
    );
  }

  @override
  void dispose() {
    _subscription?.cancel();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return Center(
      child: Column(
        mainAxisAlignment: MainAxisAlignment.center,
        children: [
          Icon(_currentIcon, size: 80, color: Colors.blue),
          SizedBox(height: 16),
          Text(
            _currentActivity,
            style: TextStyle(fontSize: 28, fontWeight: FontWeight.bold),
          ),
        ],
      ),
    );
  }
}
```

```kotlin
// SpeedHelper.kt
// 자전거 인식 보정을 위한 GPS 속도 측정

import android.location.Location
import com.google.android.gms.location.*

class SpeedHelper(private val context: Context) {

    private val fusedLocationClient = LocationServices.getFusedLocationProviderClient(context)
    private var locationCallback: LocationCallback? = null

    // 속도 기반 활동 보정
    fun getSpeedBasedActivity(speedKmh: Double, sensorActivity: String): String {
        return when {
            // 정지 상태
            speedKmh < 1.0 -> "STATIONARY"
            // 걷기 속도 범위 (1~7 km/h)
            speedKmh in 1.0..7.0 -> {
                if (sensorActivity == "RUNNING") "RUNNING" else "WALKING"
            }
            // 뛰기 속도 범위 (7~15 km/h)
            speedKmh in 7.0..15.0 -> {
                if (sensorActivity == "ON_BICYCLE") "ON_BICYCLE" else "RUNNING"
            }
            // 자전거 속도 범위 (15~35 km/h)
            speedKmh in 15.0..35.0 -> "ON_BICYCLE"
            // 그 이상은 차량
            else -> "IN_VEHICLE"
        }
    }

    // GPS 속도 모니터링 시작 (필요할 때만)
    fun startSpeedMonitoring(onSpeed: (Double) -> Unit) {
        val request = LocationRequest.Builder(
            Priority.PRIORITY_HIGH_ACCURACY, 5000 // 5초 간격
        ).setMinUpdateDistanceMeters(10f) // 10m 이상 이동 시에만
            .build()

        locationCallback = object : LocationCallback() {
            override fun onLocationResult(result: LocationResult) {
                val location = result.lastLocation ?: return
                val speedKmh = location.speed * 3.6 // m/s → km/h
                onSpeed(speedKmh)
            }
        }

        fusedLocationClient.requestLocationUpdates(
            request, locationCallback!!, context.mainLooper
        )
    }

    fun stopSpeedMonitoring() {
        locationCallback?.let {
            fusedLocationClient.removeLocationUpdates(it)
        }
    }
}
```
